## membuat partisi disk
```
cfdisk /dev/sdb

boot = 5G (Type nya diganti EFI system)

system = 100G (Type nya diganti linux filesystem)

cek partisi

lsblk
```
## setup luks
```
cryptsetup luksFormat /dev/sdb2

cryptsetup luksOpen /dev/sdb2 supernova

```
## pvcreate dan vgcreate
```
pvcreate /dev/mapper/supernova

vgcreate uas /dev/mapper/supernova

```
## LVM

## root

```
lvcreate -L 15G uas -n root

mkfs.ext4 /dev/uas/root

mount /dev/uas/root /mnt

boot

mkfs.vfat -F32 -n BOOT /dev/sdb1

mount --mkdir -o uid=0,gid=0 /dev/sdb1 /mnt/boot

````

## vars
```
lvcreate -L 15G uas -n vars

mkfs.ext4 /dev/uas/vars

mount --mkdir -o rw,nodev,nosuid,relatime

```

## Kernel parameter
```

mkdir /etc/cmdline.d

touch /etc/cmdline.d/{01-boot.conf,02-rw.conf}

echo "rd.luks.name=$(blkid -s UUID -o value /dev/sdb2)=supernova root=/dev/uas/root" > /etc/cmdline.d/01-boot.conf

```
